This configuration creates separate file-based configuration for /one and /two

Before running the commands, the properties files must be created:
ptsmc-users.properties:
    ptsmc=ptsmc

ptsmc-roles.properties:
    ptsmc=ptsmc

traveller-users.properties:
    traveller=traveller

traveller-roles.properties:
    traveller=traveller

in wildfly\bin\jboss-cli:

embed-server

/subsystem=elytron/properties-realm=ptsmcPropRealm:add(groups-attribute=groups,groups-properties={path=ptsmc-roles.properties,relative-to=jboss.server.config.dir},users-properties={path=ptsmc-users.properties,relative-to=jboss.server.config.dir,plain-text=true})
/subsystem=elytron/security-domain=ptsmcSD:add(realms=[{realm=ptsmcPropRealm,role-decoder=groups-to-roles}],default-realm=ptsmcPropRealm,permission-mapper=default-permission-mapper)
/subsystem=elytron/http-authentication-factory=ptsmc-http-auth:add(http-server-mechanism-factory=global,security-domain=ptsmcSD,mechanism-configurations=[{mechanism-name=BASIC,mechanism-realm-configurations=[{realm-name=ptsmcApplicationDomain}]}])
/subsystem=undertow/application-security-domain=ptsmcApplicationDomain:add(http-authentication-factory=ptsmc-http-auth)

/subsystem=elytron/properties-realm=travellerPropRealm:add(groups-attribute=groups,groups-properties={path=traveller-roles.properties,relative-to=jboss.server.config.dir},users-properties={path=traveller-users.properties,relative-to=jboss.server.config.dir,plain-text=true})
/subsystem=elytron/security-domain=travellerSD:add(realms=[{realm=travellerPropRealm,role-decoder=groups-to-roles}],default-realm=travellerPropRealm,permission-mapper=default-permission-mapper)
/subsystem=elytron/http-authentication-factory=traveller-http-auth:add(http-server-mechanism-factory=global,security-domain=travellerSD,mechanism-configurations=[{mechanism-name=BASIC,mechanism-realm-configurations=[{realm-name=travellerApplicationDomain}]}])
/subsystem=undertow/application-security-domain=travellerApplicationDomain:add(http-authentication-factory=traveller-http-auth)


The resulting xml looks like this:

        <subsystem xmlns="urn:wildfly:elytron:16.0" final-providers="combined-providers" disallowed-providers="OracleUcrypto">
            ...
            <security-domains>
                ...
                <security-domain name="ptsmcSD" default-realm="ptsmcPropRealm" permission-mapper="default-permission-mapper">
                    <realm name="ptsmcPropRealm" role-decoder="groups-to-roles"/>
                </security-domain>
                <security-domain name="travellerSD" default-realm="travellerPropRealm" permission-mapper="default-permission-mapper">
                    <realm name="travellerPropRealm" role-decoder="groups-to-roles"/>
                </security-domain>
            </security-domains>

            <security-realms>
                ...
                <properties-realm name="ptsmcPropRealm" groups-attribute="groups">
                    <users-properties path="ptsmc-users.properties" relative-to="jboss.server.config.dir" plain-text="true"/>
                    <groups-properties path="ptsmc-roles.properties" relative-to="jboss.server.config.dir"/>
                </properties-realm>
                <properties-realm name="travellerPropRealm" groups-attribute="groups">
                    <users-properties path="traveller-users.properties" relative-to="jboss.server.config.dir" plain-text="true"/>
                    <groups-properties path="traveller-roles.properties" relative-to="jboss.server.config.dir"/>
                </properties-realm>
            </security-realms>
            ...
            <http>
                ...
                <http-authentication-factory name="ptsmc-http-auth" security-domain="ptsmcSD" http-server-mechanism-factory="global">
                    <mechanism-configuration>
                        <mechanism mechanism-name="BASIC">
                            <mechanism-realm realm-name="ptsmcApplicationDomain"/>
                        </mechanism>
                    </mechanism-configuration>
                </http-authentication-factory>
                <http-authentication-factory name="traveller-http-auth" security-domain="travellerSD" http-server-mechanism-factory="global">
                    <mechanism-configuration>
                        <mechanism mechanism-name="BASIC">
                            <mechanism-realm realm-name="travellerApplicationDomain"/>
                        </mechanism>
                    </mechanism-configuration>
                </http-authentication-factory>
                ...
            </http>
        ...
        <subsystem xmlns="urn:jboss:domain:undertow:13.0" default-server="default-server" default-virtual-host="default-host" default-servlet-container="default" default-security-domain="other" statistics-enabled="${wildfly.undertow.statistics-enabled:${wildfly.statistics-enabled:false}}">
            ...
            <application-security-domains>
                <application-security-domain name="other" security-domain="ApplicationDomain"/>
                <application-security-domain name="ptsmcApplicationDomain" http-authentication-factory="ptsmc-http-auth"/>
                <application-security-domain name="travellerApplicationDomain" http-authentication-factory="traveller-http-auth"/>
            </application-security-domains>
        </subsystem>


In web.xml.we have:

    <security-constraint>
        <web-resource-collection>
            <web-resource-name>secure</web-resource-name>
            <url-pattern>/*</url-pattern>
        </web-resource-collection>
        <auth-constraint>
            <role-name>ptsmc/traveller</role-name>
        </auth-constraint>
    </security-constraint>

    <security-role>
        <role-name>ptsmc/traveller</role-name>
    </security-role>

    <login-config>
        <auth-method>BASIC</auth-method>
        <realm-name>ptsmc/travellerApplicationDomain</realm-name>
    </login-config>

In jboss-web.xml we have:
    <jboss-web>
        <security-domain>ptsmc/travellerApplicationDomain</security-domain>
    </jboss-web>
